                function renderSelect2() {
                  $('select').select2({
                    //placeholder: '값을 선택하세요',
                    minimumResultsForSearch: Infinity,
                    dropdownAutoWidth: true,
                  });
                  $('select:disabled').select2({
                    // placeholder: '값을 선택하세요',
                    minimumResultsForSearch: Infinity,
                    disabled: true,
                    dropdownAutoWidth: true,
                  });
                }
